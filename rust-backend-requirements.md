# Rust Backend Requirements (Diesel + SQLite)

This document specifies every Tauri command the frontend calls via `tauriInvoke`.
The backend uses **Diesel** as the ORM with a SQLite database.

---

## Migrations

Run `diesel setup` and `diesel migration generate create_initial_schema` once, then paste the schema from `sqlite-schema.sql` into the `up.sql` file.

```
migrations/
└── 2026-04-18-000000_create_initial_schema/
    ├── up.sql    ← contents of sqlite-schema.sql
    └── down.sql  ← DROP TABLE statements
```

Apply migrations at startup (see `db.rs` below).

```rust
diesel::table! {
    clients (id) {
        id         -> Text,
        name       -> Text,
        telephone  -> Text,
        created_at -> Text,
    }
}

diesel::table! {
    session_series (id) {
        id               -> Text,
        client_id        -> Text,
        recurrence_type  -> Text,
        created_at       -> Text,
    }
}

diesel::table! {
    sessions (id) {
        id            -> Text,
        client_id     -> Text,
        session_date  -> Text,
        session_time  -> Text,
        notes         -> Text,
        series_id     -> Nullable<Text>,
        created_at    -> Text,
        updated_at    -> Text,
    }
}

diesel::joinable!(sessions       -> clients (client_id));
diesel::joinable!(sessions       -> session_series (series_id));
diesel::joinable!(session_series -> clients (client_id));

diesel::allow_tables_to_appear_in_same_query!(clients, session_series, sessions);
```

## `db.rs`

```rust
// src-tauri/src/db.rs
use diesel::prelude::*;
use diesel::r2d2::{self, ConnectionManager};
use diesel_migrations::{embed_migrations, EmbeddedMigrations, MigrationHarness};

pub type DbPool = r2d2::Pool<ConnectionManager<SqliteConnection>>;

const MIGRATIONS: EmbeddedMigrations = embed_migrations!("migrations");

pub fn setup_db(db_path: &str) -> DbPool {
    let manager = ConnectionManager::<SqliteConnection>::new(db_path);
    let pool = r2d2::Pool::builder()
        .max_size(5)
        .build(manager)
        .expect("Failed to create DB pool");

    // Run pending migrations on startup
    let mut conn = pool.get().expect("Failed to get connection for migrations");
    conn.run_pending_migrations(MIGRATIONS).expect("Failed to run migrations");

    // Enable WAL mode and foreign keys for every new connection
    diesel::sql_query("PRAGMA journal_mode=WAL;").execute(&mut conn).ok();
    diesel::sql_query("PRAGMA foreign_keys=ON;").execute(&mut conn).ok();

    pool
}
```

> [!TIP]
> Store the database file in the Tauri app data directory:
> ```rust
> let db_path = app.path().app_data_dir().unwrap().join("app.db");
> let pool = setup_db(db_path.to_str().unwrap());
> ```

## `main.rs`

```rust
// src-tauri/src/main.rs
mod db;
mod models;
mod schema;
mod commands;

use db::setup_db;

fn main() {
    tauri::Builder::default()
        .setup(|app| {
            let db_path = app.path().app_data_dir()?.join("app.db");
            let pool = setup_db(db_path.to_str().unwrap());
            app.manage(pool);
            Ok(())

            commands::clients::get_clients,
            commands::clients::create_client,
```

## `commands/clients.rs`

```rust
use diesel::prelude::*;
use std::collections::HashMap;
use tauri::State;
use uuid::Uuid;
use chrono::Utc;

use crate::db::DbPool;
use crate::models::*;
use crate::schema::{clients, sessions};

type CmdResult<T> = Result<T, String>;
fn db_err(e: impl std::fmt::Display) -> String { e.to_string() }

// Helper: run a blocking Diesel closure on a thread-pool thread.
// Required because Diesel's SqliteConnection is not Send.
macro_rules! blocking {
    ($pool:expr, $closure:expr) => {{
        let pool = $pool.clone();
        tokio::task::spawn_blocking(move || {
            let mut conn = pool.get().map_err(db_err)?;
            $closure(&mut conn)
        })
        .await
        .map_err(db_err)?
    }};
}

#[tauri::command]
pub async fn get_clients(pool: State<'_, DbPool>) -> CmdResult<Vec<Client>> {
    let pool = pool.inner().clone();
    blocking!(pool, |conn: &mut SqliteConnection| {
        clients::table
            .order(clients::name.asc())
            .select(Client::as_select())
            .load(conn)
            .map_err(db_err)
    })
}

#[tauri::command]
pub async fn create_client(
    pool: State<'_, DbPool>,
    payload: ClientInsert,
) -> CmdResult<Client> {
    let pool = pool.inner().clone();
    blocking!(pool, |conn: &mut SqliteConnection| {
        let new_client = NewClient {
            id:         Uuid::new_v4().to_string(),
            name:       payload.name,
            telephone:  payload.telephone,
            created_at: Utc::now().to_rfc3339(),
        };
        diesel::insert_into(clients::table)
            .values(&new_client)
            .execute(conn)
            .map_err(db_err)?;
        clients::table
            .find(&new_client.id)
            .select(Client::as_select())
            .first(conn)
            .map_err(db_err)
    })
}

#[tauri::command]
pub async fn update_client(
    pool: State<'_, DbPool>,
    id: String,
    payload: ClientUpdate,
) -> CmdResult<Client> {
    let pool = pool.inner().clone();
    blocking!(pool, |conn: &mut SqliteConnection| {
        // Build a dynamic update — only set fields that were provided
        if let Some(name) = payload.name {
            diesel::update(clients::table.find(&id))
                .set(clients::name.eq(name))
                .execute(conn)
                .map_err(db_err)?;
        }
        if let Some(telephone) = payload.telephone {
            diesel::update(clients::table.find(&id))
                .set(clients::telephone.eq(telephone))
                .execute(conn)
                .map_err(db_err)?;
        }
        clients::table
            .find(&id)
            .select(Client::as_select())
            .first(conn)
            .map_err(db_err)
    })
}

#[tauri::command]
pub async fn delete_client(pool: State<'_, DbPool>, id: String) -> CmdResult<()> {
    let pool = pool.inner().clone();
    blocking!(pool, |conn: &mut SqliteConnection| {
        diesel::delete(clients::table.find(&id))
            .execute(conn)
            .map(|_| ())
            .map_err(db_err)
    })
}

#[tauri::command]
pub async fn get_session_counts(pool: State<'_, DbPool>) -> CmdResult<HashMap<String, u32>> {
    let pool = pool.inner().clone();
    blocking!(pool, |conn: &mut SqliteConnection| {
        // Diesel doesn't support GROUP BY + COUNT natively without a custom type,
        // so we use sql_query here.
        #[derive(QueryableByName)]
        struct Row {
            #[diesel(sql_type = diesel::sql_types::Text)]
            client_id: String,
            #[diesel(sql_type = diesel::sql_types::BigInt)]
            count: i64,
        }

        let rows = diesel::sql_query(
            "SELECT client_id, COUNT(*) as count FROM sessions GROUP BY client_id"
        )
        .load::<Row>(conn)
        .map_err(db_err)?;

        Ok(rows.into_iter().map(|r| (r.client_id, r.count as u32)).collect())
    })
}
```

## `commands/sessions.rs`

```rust
use diesel::prelude::*;
use tauri::State;
use uuid::Uuid;
use chrono::{Utc, NaiveDate, Months};

use crate::db::DbPool;
use crate::models::*;
use crate::schema::{clients, sessions, session_series};

type CmdResult<T> = Result<T, String>;
fn db_err(e: impl std::fmt::Display) -> String { e.to_string() }

fn now() -> String { Utc::now().to_rfc3339() }

fn advance_date(date_str: &str, recurrence: &str) -> String {
    let date = NaiveDate::parse_from_str(date_str, "%Y-%m-%d").expect("invalid date");
    let next = match recurrence {
        "weekly"   => date + chrono::Duration::days(7),
        "biweekly" => date + chrono::Duration::days(14),
        "monthly"  => date.checked_add_months(Months::new(1)).unwrap(),
        _          => panic!("unknown recurrence type"),
    };
    next.format("%Y-%m-%d").to_string()
}

#[tauri::command]
pub async fn get_sessions(pool: State<'_, DbPool>) -> CmdResult<Vec<SessionWithClient>> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;
        let rows: Vec<(Session, Client)> = sessions::table
            .inner_join(clients::table)
            .order((sessions::session_date.desc(), sessions::session_time.asc()))
            .select((Session::as_select(), Client::as_select()))
            .load(&mut conn)
            .map_err(db_err)?;

        Ok(rows.into_iter().map(|(s, c)| SessionWithClient {
            id:           s.id,
            client_id:    s.client_id,
            session_date: s.session_date,
            session_time: s.session_time,
            notes:        s.notes,
            series_id:    s.series_id,
            created_at:   s.created_at,
            updated_at:   s.updated_at,
            client:       c,
        }).collect())
    })
    .await
    .map_err(db_err)?
}

#[tauri::command]
pub async fn get_sessions_by_client(
    pool: State<'_, DbPool>,
    client_id: String,
    before_date: String,
    exclude_id: Option<String>,
    limit: i64,
) -> CmdResult<Vec<Session>> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;
        let mut query = sessions::table
            .filter(sessions::client_id.eq(&client_id))
            .filter(sessions::session_date.lt(&before_date))
            .order(sessions::session_date.desc())
            .limit(limit)
            .select(Session::as_select())
            .into_boxed();

        if let Some(ref eid) = exclude_id {
            query = query.filter(sessions::id.ne(eid));
        }

        query.load(&mut conn).map_err(db_err)
    })
    .await
    .map_err(db_err)?
}

#[tauri::command]
pub async fn create_session(
    pool: State<'_, DbPool>,
    payload: SessionInsert,
) -> CmdResult<Session> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;
        let ts = now();
        let new_session = NewSession {
            id:           Uuid::new_v4().to_string(),
            client_id:    payload.client_id,
            session_date: payload.session_date,
            session_time: payload.session_time,
            notes:        payload.notes,
            series_id:    payload.series_id,
            created_at:   ts.clone(),
            updated_at:   ts,
        };
        diesel::insert_into(sessions::table)
            .values(&new_session)
            .execute(&mut conn)
            .map_err(db_err)?;
        sessions::table
            .find(&new_session.id)
            .select(Session::as_select())
            .first(&mut conn)
            .map_err(db_err)
    })
    .await
    .map_err(db_err)?
}

#[tauri::command]
pub async fn update_session(
    pool: State<'_, DbPool>,
    id: String,
    payload: SessionUpdate,
) -> CmdResult<Session> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;
        let target = sessions::table.find(&id);

        if let Some(v) = payload.client_id    { diesel::update(target).set(sessions::client_id.eq(v)).execute(&mut conn).map_err(db_err)?; }
        if let Some(v) = payload.session_date { diesel::update(target).set(sessions::session_date.eq(v)).execute(&mut conn).map_err(db_err)?; }
        if let Some(v) = payload.session_time { diesel::update(target).set(sessions::session_time.eq(v)).execute(&mut conn).map_err(db_err)?; }
        if let Some(v) = payload.notes        { diesel::update(target).set(sessions::notes.eq(v)).execute(&mut conn).map_err(db_err)?; }
        if let Some(v) = payload.series_id    { diesel::update(target).set(sessions::series_id.eq(v)).execute(&mut conn).map_err(db_err)?; }

        diesel::update(target)
            .set(sessions::updated_at.eq(now()))
            .execute(&mut conn)
            .map_err(db_err)?;

        sessions::table.find(&id).select(Session::as_select()).first(&mut conn).map_err(db_err)
    })
    .await
    .map_err(db_err)?
}

#[tauri::command]
pub async fn delete_session(pool: State<'_, DbPool>, id: String) -> CmdResult<()> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;
        diesel::delete(sessions::table.find(&id))
            .execute(&mut conn)
            .map(|_| ())
            .map_err(db_err)
    })
    .await
    .map_err(db_err)?

#[tauri::command]
pub async fn create_session_series(
    pool: State<'_, DbPool>,
    payload: CreateSeriesPayload,
) -> CmdResult<SessionSeries> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;
        let series = SessionSeries {
            id:              Uuid::new_v4().to_string(),
            client_id:       payload.client_id.clone(),
            recurrence_type: payload.recurrence_type.clone(),
            created_at:      now(),
        };
        diesel::insert_into(session_series::table)
            .values(&series)
            .execute(&mut conn)
            .map_err(db_err)?;

        let mut current_date = payload.start_date.clone();
        for i in 0..payload.num_sessions {
            let ts = now();
            let new_session = NewSession {
                id:           Uuid::new_v4().to_string(),
                client_id:    payload.client_id.clone(),
                session_date: current_date.clone(),
                session_time: payload.start_time.clone(),
                notes:        if i == 0 { payload.notes.clone().unwrap_or_default() } else { String::new() },
                series_id:    Some(series.id.clone()),
                created_at:   ts.clone(),
                updated_at:   ts,
            };
            diesel::insert_into(sessions::table)
                .values(&new_session)
                .execute(&mut conn)
                .map_err(db_err)?;
            current_date = advance_date(&current_date, &payload.recurrence_type);
        }

        Ok(series)
    })
    .await
    .map_err(db_err)?
}

#[tauri::command]
pub async fn extend_session_series(
    pool: State<'_, DbPool>,
    payload: ExtendSeriesPayload,
) -> CmdResult<()> {
    let pool = pool.inner().clone();
    tokio::task::spawn_blocking(move || {
        let mut conn = pool.get().map_err(db_err)?;

        let series: SessionSeries = session_series::table
            .find(&payload.series_id)
            .select(SessionSeries::as_select())
            .first(&mut conn)
            .map_err(db_err)?;

        let mut current_date = advance_date(&payload.from_date, &payload.recurrence_type);
        for _ in 0..payload.num_sessions {
            let ts = now();
            let new_session = NewSession {
                id:           Uuid::new_v4().to_string(),
                client_id:    series.client_id.clone(),
                session_date: current_date.clone(),
                session_time: payload.session_time.clone(),
                notes:        String::new(),
                series_id:    Some(payload.series_id.clone()),
                created_at:   ts.clone(),
                updated_at:   ts,
            };
            diesel::insert_into(sessions::table)
                .values(&new_session)
                .execute(&mut conn)
                .map_err(db_err)?;
            current_date = advance_date(&current_date, &payload.recurrence_type);
        }
        Ok(())
    })
    .await
    .map_err(db_err)?
}
```

## Notes

> [!TIP]
> **`SqliteConnection` is not `Send`** — Diesel's SQLite driver uses a non-thread-safe connection type. Every `#[tauri::command]` that touches the DB must wrap its closure in `tokio::task::spawn_blocking`.

> [!TIP]
> **Partial updates with Diesel** — Diesel doesn't support a single variable-length `SET` clause natively without the `AsChangeset` macro. The pattern above issues multiple small `UPDATE` statements per changed field. For cleaner code you can implement `AsChangeset` on `ClientUpdate` / `SessionUpdate` using `#[diesel(treat_none_as_null = true)]`.

> [!IMPORTANT]
> **Enable foreign keys** — SQLite does not enforce foreign keys by default. Add `PRAGMA foreign_keys = ON` after each new connection is created (hooking into the r2d2 `ConnectionCustomizer` is the cleanest way).

> [!NOTE]
> **Diesel CLI** — Install with `cargo install diesel_cli --no-default-features --features sqlite`. Run `diesel setup` to create the database and run migrations. Run `diesel print-schema > src/schema.rs` whenever the schema changes.
